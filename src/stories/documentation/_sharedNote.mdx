import { Canvas, Meta, Story, Source } from "@storybook/blocks";
import { initTabs } from "./storybook-documentation.js";

/\*\*

- SharedNote component.
-
- Renders a shared note section in documentation stories.
-
- @param {object} defaultdata - The default data to be displayed in the JSON tab.
- @param {object} designResources - Design resource links or assets related to the note to be displayed in Design Resources tab.
- @param {string} componentId - Id value of associated component to be fed into code block.
- @param {React.ReactNode} children - Content to be rendered inside the shared note.
  \*/

export const SharedNote = ({ defaultdata, designResources, componentId, children }) => (
  <>

    <div className="tab">
      <button className="tablinks active" data-targetid="code-doc">
        Code
      </button>
      <button className="tablinks" data-targetid="json-doc">
        JSON
      </button>
      <button className="tablinks" data-targetid="design-doc">
        Design Resources
      </button>
      <button className="tablinks" data-targetid="instructions-doc">
        Usage Instructions
      </button>
    </div>

    <div id="code-doc" className="tabcontent active">
      <h3>Component code</h3>
      <Source language="html" id={componentId} />
    </div>

    <div id="json-doc" className="tabcontent" >
      <h3>Example Component data</h3>

      <Source language="json" code={JSON.stringify(defaultdata, null, 2)} />
    </div>

    <div id="design-doc" className="tabcontent">
      <h3>Design resources</h3>
      <ul>
        {designResources.map((item,index) => (
          <li key={`design-resource${index}`}><a
            href={item?.url}
            target="_top"
            rel="nofollow"
            class="sbdocs sbdocs-a css-ajfpqr"
          >
            {item?.text}
          </a></li>
        ))}
      </ul>

    </div>

    <div id="instructions-doc" className="tabcontent">
      {children}
    </div>
    <script>initTabs(); // Call the function immediately</script>

  </>
);

;
